// create a db schema in mongodb in a separate file

const mongoose = require('mongoose');

const FoodSchema = new mongoose.Schema({
    foodName:{
        type:String,
        required:true
    },
    daysSinceIAte:{
        type:Number,
        required:true
    }
})

const Food = mongoose.model("FoodData",FoodSchema)

module.exports = Food;
